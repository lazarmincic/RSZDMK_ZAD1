char ocitaj_taster()
{
  char taster = 0;
  unsigned char row;
  
  MCUCR &= ~0x10;
  PORTD = 0x0f; 
  
  for(row = 0x80; row != 0x08; row >>= 1)
  {
    DDRD = row;
	_delay_ms(2);
    
    switch(row | ((~PIND) & 0x0f))
    {
      case 0x88:
      	taster = '1'; break;
      case 0x84:
      	taster = '2'; break;
      case 0x82:
      	taster = '3'; break;
      case 0x81:
      	taster = 'A'; break;
		
	  case 0x48:
      	taster = '4'; break;
      case 0x44:
      	taster = '5'; break;
      case 0x42:
      	taster = '6'; break;
      case 0x41:
      	taster = 'B'; break;
		
	  case 0x28:
      	taster = '7'; break;
      case 0x24:
      	taster = '8'; break;
      case 0x22:
      	taster = '9'; break;
      case 0x21:
      	taster = 'C'; break;
		
	  case 0x18:
      	taster = '*'; break;
      case 0x14:
      	taster = '0'; break;
      case 0x12:
      	taster = '#'; break;
      case 0x11:
      	taster = 'D'; break;
    }
  }
  
  return taster;
}

 char unos [4]={0,0,0,0}; //unos

int main()
{
  	
 
  	lcd.begin(16, 2);
	bool start = 0;
  	unsigned char redosled [4]; //komb. sa pocetka
  	lcd.createChar (0,pik);
    lcd.createChar (1,tref);
    lcd.createChar (2,herc);
    lcd.createChar (3,karo);
    bool win = 0;
	while(1)
    { 
       
 	 uint32_t seed = 0;
	  char taster;
      lcd.clear();
      lcd.home();
      lcd.print("Pritisni *");
      while(ocitaj_taster()!='*')
        seed++;
      randomSeed(seed);
      taster = ocitaj_taster();
       
      start = 1;
      if (start)
      {
        lcd.clear();
        for (int i=0;i<4;i++)
        {
         	redosled[i] = rand() % 4 ;
            lcd.write (byte(redosled[i]));
        }
        _delay_ms(2000);
        lcd.clear();
        lcd.blink();
        lcd.print("----");
        lcd.home();
      
         char brojac = 0; // koji znak
        int kursor = 0; // mesto kursora
        bool kraj=0; // da li su svi simb. uneti
        while(kraj==0)
        {
          taster = ocitaj_taster();
          
          switch (taster)
          {
           case '2':
              lcd.setCursor(kursor,0);
              lcd.write(byte(brojac));
           	  lcd.setCursor(kursor,0);
              unos[kursor]=brojac;
              brojac++;
              brojac&=0x03;
              break;
           case '8':
              lcd.setCursor(kursor,0);
              lcd.write(byte(brojac));
              lcd.setCursor(kursor,0);
              unos[kursor]=brojac;
              brojac--;
              brojac&=0x03;
              break;
           case '4':
              kursor--;
              kursor&=0x03;
              lcd.setCursor(kursor,0);
              break;
            case '6':
              kursor++;
              kursor&=0x03;
              lcd.setCursor(kursor,0);
              break;
            case '5':
              
              for (int i=0;i<4;i++)
                  if (unos[i]== 0)
                    kraj = 0;
                kraj = 1;
              if (kraj)
              {
                  lcd.noBlink();
                  win = 1;
                  for (int i=0;i<4;i++)
                  {
                      if (redosled[i]!=unos[i])
                        win = 0;
                  }
                  lcd.setCursor(0,1);
                  for (int i =0;i<4;i++)
                      lcd.write(byte(redosled[i]));
                  if (win)
                  {
                    lcd.setCursor(7,1);
                    lcd.print("To lave!");
                    while(1);
                  }
                  else
                  {
                    lcd.setCursor(7,0);
                    lcd.print("Nemas");
                    lcd.setCursor(7,1);
                    lcd.print("pojma :P");
                    while(1);
                  }
              }
          }
          while(ocitaj_taster()==taster);
        }
      }
      
    }
  
  	return 0;
}
